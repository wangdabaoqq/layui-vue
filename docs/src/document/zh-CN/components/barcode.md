::: title 显示编码
:::

::: demo 通过 `displayValue` 属性，显示二维码文字，使用 `text` 属性覆盖默认显示的值。

<template>
  <lay-barcode :value="value" :displayValue="true"/>
</template>

<script>
import { ref } from 'vue'

export default {
  setup() {

    const value = ref("6666");

    return {
      value
    }
  }
}
</script>

:::
