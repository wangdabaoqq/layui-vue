<script>
import { ref } from 'vue'

export default {
  setup() {
    const replay = (e, animClass) => {
      const el = e.currentTarget;
      const targetClass = el.classList[1];
      if(!targetClass){
        el.classList.add(animClass);
      }else{
        el.classList.remove(targetClass);
        setTimeout(() => {
          el.classList.add(targetClass);
        },100) 
      }
    }
    return {
      replay
    }
  }
}
</script>
