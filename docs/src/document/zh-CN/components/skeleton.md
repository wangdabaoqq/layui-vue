    const loading = ref(true);
    return {

    }
  }

::: title Skeleton 属性
:::

::: table

| 名称     | 描述     | 类型      | 默认值  | 可选值         |
| -------- | -------- | --------- | ------- | -------------- |
| loading  | 是否显示 | `boolean` | `false` | `true` `false` |
| rows     | 显示行数 | `number`  | --      | --             |
| animated | 是否动画 | `boolean` | `true`  | `true` `false` |
| type     | 展示类型 | `string`  | `p`     | `p` `image`    |

:::

::: title Skeleton 插槽
:::

::: table

| 名称     | 描述        |
| -------- | ---------- |
| default  | 默认插槽    |
| skeleton | 自定义插槽  |
